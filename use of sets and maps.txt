int findanswer(int k,stack<int>s)
{
    int freq=0;
    while(k>0)
    {
        int k1=s.top();
        s.pop();
        --k;
        freq+=k1;
    }
    return freq;
}

               // cout<<it1.first<<" "<<it1.second<<endl;
            }
            while(q--)

                int k;

                int ans;
